015        /**
016        *Constructor el cual si se le pasa un argumento que se salga
017        *de los parametros marcados hará que salga una DateException.
018        *
019        *-El año siempre será correcto
020        *-El mes ha de estar entre el 1 y el 12, si no dirá que no es valido
021        *-El día no puede ser <1 si no dirá que el día no es valido,
022        *luego comprobará según el mes los días que puede tener 
023        */
024        
025        public Date(int day, int month, int year)throws DateException{
026
027                this.year=year;
028
029                if(month<1||month>12){
030
031                        throw new DateException("mes "+month+" no valido "+"valores posibles entre 1 y 12");
032                }
033                else{
034
035                        this.month=month;
036                }
037
038                if(day<1){
039                throw new DateException("dia "+day+ " no valido "+" solo se pueden dias mayores que 1");
040                }
041                else{
042                        switch(this.month){
043                        case 1:
044                        case 3:
045                        case 5:
046                        case 7:
047                        case 8:
048                        case 10:
049                        case 12:
050                                if(day>31){
051                                throw new DateException("dia "+day+" no valido en este mes");
052                                }
053                                else{
054                                this.day=day;
055                                }
056                        break;
057                        case 4:
058                        case 6:
059                        case 9:
060                        case 11:
061                                if(day>30){
062                                throw new DateException("dia "+day+" no valido en este mes");
063                                }
064                                else{
065                                this.day=day;
066                                }
067                        break;
068                        case 2:
069                                
070                                if(((year%400==0)||(year%4==0 && year%100!=0))&&day>29){
071                                throw new DateException("dia "+day+" no valido en el mes de febrero");
072                                }
073                                else if (day>28){
074                                this.day=day;
075                                }
076                        break;
077                        
078                        }
079                }
080}
