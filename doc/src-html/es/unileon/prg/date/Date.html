001package es.unileon.prg.date;
002
003public class Date {
004
005        private int day;
006        private int month;
007        private int year;
008
009        public Date(int day, int month, int year) throws DateException{
010                this.year = year;
011                if (month < 1 || month > 12) {
012                        throw new DateException("Mes " + month + " no valido" +
013                                        " Valores posibles entre 1 y 12.");
014                } else {
015                        this.month = month;
016                }
017                //TODO falta comprobar el dia
018                this.day = day;
019        }
020
021        @Override
022        public String toString() {
023                return this.day + "/" + this.month + "/" + this.year;
024        }
025
026}
